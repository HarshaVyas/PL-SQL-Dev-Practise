---LEFT OUTER JOIN
select * from CUSTOMER_DETAILS C LEFT OUTER JOIN Order_Details O on C.CUST_ID=O.CUSTOMER_ID

---RIGHT OUTER JOIN
select * from CUSTOMER_DETAILS C RIGHT OUTER JOIN Order_Details O on C.CUST_ID=O.CUSTOMER_ID

--- INNER JOIN
select * from CUSTOMER_DETAILS C INNER JOIN Order_Details O on C.CUST_ID=O.CUSTOMER_ID

---FULL OUTER JOIN
